import { useState } from 'react'

export const InteractiveCounter = () => {
  const [count, setCount] = useState(0)
  return (
    <div style={{
      padding: '1rem',
      border: '2px solid var(--ui-color-primary, #3b82f6)',
      borderRadius: '0.5rem',
      margin: '1rem 0',
      textAlign: 'center',
      backgroundColor: 'var(--ui-color-background-secondary, #f8f9fa)'
    }}>
      <h4>Interactive Counter Component</h4>
      <p>Current count: <strong>{count}</strong></p>
      <button
        onClick={() => setCount(count + 1)}
        style={{
          padding: '0.5rem 1rem',
          backgroundColor: 'var(--ui-color-primary, #3b82f6)',
          color: 'white',
          border: 'none',
          borderRadius: '0.25rem',
          cursor: 'pointer',
          marginRight: '0.5rem'
        }}
      >
        Increment
      </button>
      <button
        onClick={() => setCount(0)}
        style={{
          padding: '0.5rem 1rem',
          backgroundColor: 'var(--ui-color-text-secondary, #4a4a4a)',
          color: 'white',
          border: 'none',
          borderRadius: '0.25rem',
          cursor: 'pointer'
        }}
      >
        Reset
      </button>
    </div>
  )
}

MDX is a format that lets you seamlessly write JSX in your Markdown documents. You can import components, such as interactive charts or alerts, and embed them within your content. This makes writing long-form content with components a blast! ðŸš€
